138{
139 int valorMaior, valorMenor, porColocar;

143 // - valor atual no outro lado, o maior

145 valorMenor = totalEsquerda;
146 if (valorMaior < valorMenor) {
147 porColocar = valorMaior;
148 valorMaior = valorMenor;
149 valorMenor = porColocar;
150 }

152 estado[0] = porColocar;
153 estado[1] = valorMaior;
