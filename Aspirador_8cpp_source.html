97// o problema é simples, é possível uma heurística perfeita:
98// dá sempre o custo até à solução objetivo ótimo
99int CAspirador::Heuristica(void) {
100 // o aspirador tem de se mover para a salas mais extremas, e limpar todas as casas sujas
101 int sujas = 0, max = -1, min = -1;
102 for (int i = 0; i < salas.Count(); i++)
103 if (salas[i] == '*') {
104 sujas++;
105 if (min < 0 || i < min)
106 min = i;
107 if (max < 0 || i > max)
108 max = i;
109 }
110 if (sujas == 0)
111 return heuristica = 0;
112 return heuristica = sujas + // movimentos a aspirar as casas sujas
113 (abs(aspirador - min) < abs(aspirador-max) ? // ver qual o extermo mais próximo
114 abs(aspirador - min) : // movimentos a ir para o extermo mais próximo
115 abs(aspirador-max)) +
116 max - min; // após estar nesse extremo, ainda vai ao outro extremo
117}
